{% comment %} Renders bilingual excerpts when excerpt is a hash with `en` and `pt`. Falls back to plain string excerpts. Usage: {% include bilingual-excerpt.html excerpt=item.excerpt words=28 %} {% endcomment %} {% if include.excerpt %} {% if include.excerpt.en %} {{ include.excerpt.en }} {{ include.excerpt.pt }} {% else %} {% assign excerpt_text = include.excerpt %} {% if include.words %} {{ excerpt_text | strip_html | truncatewords: include.words }} {% else %} {{ excerpt_text | strip_html }} {% endif %} {% endif %} {% endif %}
